  value: T,

): T {

好的，基于你引用的 useDebounce，我给它加上了 cancel 方法和 immediate 参数：

typescript 
export function useDebounce<T>(
  value: T,
  delay = 300,
  options?: { immediate?: boolean }
) {
  const [debounced, setDebounced] = useState(value);
  const timerRef = useRef<NodeJS.Timeout>();

  const cancel = useCallback(() => {
    if (timerRef.current) clearTimeout(timerRef.current);
  }, []);

  useEffect(() => {
    if (options?.immediate && !timerRef.current) {
      setDebounced(value);
    }
    timerRef.current = setTimeout(
      () => setDebounced(value), delay
    );
    return cancel;
  }, [value, delay]);

  return { value: debounced, cancel };
}

主要改动：
• 返回值从 T 改为 { value, cancel }
• 新增 immediate 选项：首次调用立即执行
• cancel() 方法清除 pending timer
