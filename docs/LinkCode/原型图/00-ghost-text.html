import { User, UserRole, UserPreferences } from '../models/user';
import { apiClient } from './apiClient';
import { cache } from '../utils/cache';

/** 用户服务 - 处理用户 CRUD 操作 */
export class UserService {
  private baseUrl = '/api/v1/users';

  /** 根据 ID 获取用户信息 */
  async getUserById(id: string): Promise<User | null> {
    const cached = cache.get<User>(`user:${id}`);
    if (cached) return cached;

    const response = await apiClient.get<User>(`${this.baseUrl}/${id}`);
    cache.set(`user:${id}`, response.data, { ttl: 300 });
        return response.data;
      }

    return response.data;
  }

  /** 更新用户偏好设置 */
  async updatePreferences(
    userId: string,
    prefs: Partial<UserPreferences>
  ): Promise<UserPreferences> {
    const response = await apiClient.patch<UserPreferences>(
      `${this.baseUrl}/${userId}/preferences`,
      prefs
    );
    cache.invalidate(`user:${userId}`);
    return response.data;
  }

  /** 获取用户列表 */
  async listUsers(page = 1, limit = 20): Promise<User[]> {
    const res = await apiClient.get(`${this.baseUrl}?page=${page}&limit=${limit}`);
    return res.data;
  }
}
